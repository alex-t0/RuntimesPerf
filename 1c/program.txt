Функция РассчитатьОценку(Число)
    ЧислоДляПодсчета = Число;
    КоличествоЦифр = 0;
    МассивЦифр = Новый Массив;
    
    Пока ЧислоДляПодсчета > 0 Цикл
        ЧислоДляПодсчета = Цел(ЧислоДляПодсчета / 10);
        МассивЦифр.Добавить(ЧислоДляПодсчета);
        КоличествоЦифр = КоличествоЦифр + 1;
    КонецЦикла;
    
    Если КоличествоЦифр = 0 Тогда
        Возврат 0;
    КонецЕсли;
    
    БазовыйКорень = МассивЦифр[МассивЦифр.Количество() - 1];
    Экспонента = КоличествоЦифр - 1;
    
    Если КоличествоЦифр % 2 = 0 Тогда
        БазовыйКорень = МассивЦифр[МассивЦифр.Количество() - 2];
        Экспонента = КоличествоЦифр - 2;
    КонецЕсли;
    
    Если БазовыйКорень < 10 Тогда
        Возврат 2 * Pow(10, Экспонента / 2);
    КонецЕсли;
    
    Возврат 6 * Pow(10, Экспонента / 2);

Функция КвадратныйКорень(Число)
    Если Число = 0 Тогда
        Возврат 0;
    КонецЕсли;
    
    ТекущаяТочность = 1;
    Оценка = РассчитатьОценку(Число);
	
	Точность = 0.00000001;
	
    Пока ТекущаяТочность > Точность Цикл
        Оценка = (Число / Оценка + Оценка) / 2;
        ТекущаяТочность = Abs(Оценка * Оценка - Число);
    КонецЦикла;
    
    Возврат Оценка;
